108 /**
109  * Executes a callback with information as to the current location.
110  * @param {{interval: int}} settings Has an optional member, interval, that specified how many seconds
111  *                                  between calls to callback.
112  * @param {?function(({coords: {lat: double, lon: double}, status: string, message: string}|{status: string, message: string}))}
113  *              receiverCallback A callback function to get the results.  Must take one argument <code>data</code>.
114  * @param {Object} [callback.data]
115  * <ul>
116  * <li>status {string}: "success" for sucess, "failure" for failure.</li>
117  * <li>message {string}: Empty on success, otherwise populated with the error message</li>
118  * <li>coords {Object}: An object with a double <code>lat</code> and a double <code>lon</code>.</li>
119  * </ul>
120  * @param {?function(({coords: {lat: double, lon: double}, status: string, message: string}|{status: string, message: string}))}
121  *              callback A callback function to get the response from issuing the call.  Must take one argument.
122  */
123 Mono.Location.GetCurrentLocation =  function(settings, receiverCallback, callback)
124 {
125     if(Mono.IsNative())
126     {
127         var configParams = Ozone.util.parseWindowNameData();
128         if(configParams !== null)  {
129             var callbackId = configParams.id + "LOCATION_TIME" + Mono.Callbacks.generateGuid();
130             Mono.Callbacks.registerCallback(this, receiverCallback, callbackId);
131             return Mono.MakeRestCall("location/current", $.extend({"callback": callbackId}, settings), callback);
132         }
133     }
134     else
135     {
136         if (navigator.geolocation)
137         {
138             navigator.geolocation.getCurrentPosition(receiverCallback,function(msg){console.log(msg);});
139         }
140         else
141         {
142             console.log("geolocation is not available.");
143         }
144     }
145 }

185 /**
186  * Disables updates for geolocation data.
187  * @param {{type: string}} settings An object with a member "type."  Type must be "time" to unregister from
188  *                                  GetCurrentLocation updates and "distance" to unregister from RegisterForUpdate
189  *                                  updates.
190  */
191 Mono.Location.DisableLocationUpdates = function(settings)
192 {
193     if(Mono.IsNative())
194     {
195         data['type'] = settings.type;
196         var configParams = Ozone.util.parseWindowNameData();
197         var callbackId;
198         if(settings.type == "time") {
199              callbackId = configParams.id + "LOCATION_TIME";
200         }
201         else if(settings.type == "distance") {
202              callbackId = configParams.id + "LOCATION_DISTANCE";
203         }
204         else {
205              callbackId = configParams.id + "LOCATION";
206         }
207 
208         Mono.Callbacks.RemoveAllThatStartWith(callbackId);
209         return Mono.MakeRestCall("location/disableUpdates", data, null);
210     }
211     else
212     {
213         // Do nothing
214     }
215 }

584 Mono.EventBus = function(){
585 	var _eventMap = {};
586 	function _generateId() {
587 		function s4() {
588 	    	return Math.floor((1 + Math.random()) * 0x10000)
589 	        		   .toString(16)
590 	               	.substring(1);
591 	 		   }
592 	  		 return s4() + s4() + '-' + s4() + '-' + s4() + '-' +
593 	         	s4() + '-' + s4() + s4() + s4();
594 	};
595 
596 	function _register(f, name){
597 		var id = name || _generateId();
598 		if(!_eventMap[id]) _eventMap[id] = [];
599 		_eventMap[id].push(f);
600 		return id;
601 	};
602 
603 	function _runEvents(id, data){
604 		if(_eventMap[id] && _eventMap[id].length){
605 		  for(var i=0; i<_eventMap[id].length; i++){
606 			  _eventMap[id][i](data);
607 		  }
608 		}
609 	};
610 
611 	return {
612 		generateId: _generateId,
613 		register: _register,
614 		runEvents: _runEvents
615 	};
616 }();

799 /**
800  * Gets data from a URL.  If done on a native device, the data is cached.
801  * If the device goes offline, the data will be retrieved from the cache.
802  * @param {string} url The URL to retrieve.
803  * @param {string} method Either GET or POST
804  * @param {Object} params Parameters to pass to a URL via POST.  Should be a regular JSON object.
805  * @param {string} callback A callback function to get the results.
806  * <ul>
807  * <li>status {string}: "success" for sucess, "failure" for failure.</li>
808  * <li>message {string}: Empty on success, otherwise populated with the error message</li>
809  * <li>data {string}: The data retrieved from the URL.
810  * </ul>
811  */
812 Mono.Caching.GetDataFromUrl = function(url, method, params, callback)
813 {
814     if(Mono.IsNative())
815     {
816         var data = new Object();
817         data.url = url;
818         data.method = method;
819         if(params != null)
820         {
821             data.params = params;
822         }
823         return Mono.MakeRestCall("caching/getDataFromUrl", data, callback);
824     }
825     else
826     {
827         xmlhttp = new XMLHttpRequest();
828 
829         xmlhttp.onreadystatechange = function()
830         {
831             if(xmlhttp.readyState == 4 && xmlhttp.status == 200)
832             {
833                 var data = new Object();
834 
835                 data.data = xmlhttp.responseText;
836                 data.status = "success";
837 
838                 if(callback)
839                 {
840                     callback(data);
841                 }
842             }
843             else if(xmlhttp.readyState == 4)
844             {
845                 var data = new Object();
846 
847                 data.status = "failure";
848                 data.message = "Error getting data.";
849                 if(callback)
850                 {
851                     callback(data);
852                 }
853             }
854         }
855 
856         xmlhttp.onerror = function()
857         {
858             console.log("Error: " + xmlhttp);
859             var data = new Object();
860 
861             data.status = "failure";
862             data.message = "Error getting data.";
863             if(callback)
864             {
865                 callback(data);
866             }
867         }
868 
869         xmlhttp.open(method, url, true);
870 
871         var formData = new FormData();
872 
873         for(var key in params)
874         {
875             if(params.hasOwnProperty(key))
876             {
877                 formData.append(key, params[key]);
878             }
879         }
880 
881         xmlhttp.send(formData);
882     }
